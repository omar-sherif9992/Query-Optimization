-- Query 7 view table of reserves with bid = 103

create MATERIALIZED VIEW query_7
as
select r.sid
from reserves r
where  r.bid =103  ;


select s.sname
from sailors s
where exists (select R.sid
              from query_7 R
              where s.sid =R.sid);

  -- Query 8 view table of reserves sids that have red boats

create MATERIALIZED VIEW query_8
as
select r1.sid
 from (select  r.sid
       from reserves r
       where exists (select * from boat b where r.bid=b.bid  and color='red'  ) ) as r1 ;

---------------------Query 9 Optimized  -----------------  
select s.sname
from sailors s
where exists
        (
         select rTotal.sid
            from (select r1.sid
             from
                (select r.sid
                from reserves r
                 where  exists
                    (select bid
                     from boat b
                     where color = 'green' and r.bid =b.bid )
                )as r1
             inner join
                (select r.sid
                from reserves r
                 where  exists
                    (select bid
                     from boat b
                     where color = 'red' and r.bid =b.bid )
                ) as r2
             on r2.sid = r1.sid  ) as rTotal
          where rTotal.sid=s.sid
)
